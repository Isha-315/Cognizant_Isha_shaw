public class FinancialForecast {

    public static double futureValue(double initialValue, double growthRate, int years) {
        if (years == 0) return initialValue;
        return futureValue(initialValue, growthRate, years - 1) * (1 + growthRate);
    }

    public static double futureValue(double initialValue, double[] growthRates, int year) {
        if (year == 0) return initialValue;
        return futureValue(initialValue, growthRates, year - 1) * (1 + growthRates[year - 1]);
    }

    public static double futureValueIterative(double initialValue, double growthRate, int years) {
        double fv = initialValue;
        for (int i = 0; i < years; i++) {
            fv *= (1 + growthRate);
        }
        return fv;
    }

    public static void main(String[] args) {
        double initialInvestment = 1000;
        double growthRate = 0.05;
        int years = 5;

        double fv = futureValue(initialInvestment, growthRate, years);
        System.out.println("Future Value after " + years + " years: $" + fv);

        double[] pastGrowthRates = {0.05, 0.06, 0.04, 0.05, 0.03};
        double fvVariable = futureValue(initialInvestment, pastGrowthRates, pastGrowthRates.length);
        System.out.println("Future Value with variable growth: $" + fvVariable);

        double fvIterative = futureValueIterative(initialInvestment, growthRate, years);
        System.out.println("Future Value (iterative) after " + years + " years: $" + fvIterative);
    }
}
